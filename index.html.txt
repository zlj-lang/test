    <script>
        // 照片数据管理
        let photos = [];
        const STORAGE_KEY = 'countrysidePhotos';
        
        // 初始化页面
        document.addEventListener('DOMContentLoaded', function() {
            loadPhotos();
            updateStats();
            
            // 绑定事件
            document.getElementById('uploadBtn').addEventListener('click', function() {
                document.getElementById('fileInput').click();
            });
            
            document.getElementById('uploadArea').addEventListener('click', function() {
                document.getElementById('fileInput').click();
            });
            
            document.getElementById('fileInput').addEventListener('change', handleFileUpload);
            
            document.getElementById('clearAllBtn').addEventListener('click', clearAllPhotos);
        });
        
        // 加载照片
        function loadPhotos() {
            const storedPhotos = localStorage.getItem(STORAGE_KEY);
            if (storedPhotos) {
                photos = JSON.parse(storedPhotos);
                displayPhotos();
            }
        }
        
        // 保存照片
        function savePhotos() {
            localStorage.setItem(STORAGE_KEY, JSON.stringify(photos));
            updateStats();
        }
        
        // 显示照片
        function displayPhotos() {
            const container = document.getElementById('photosContainer');
            const emptyState = document.getElementById('emptyState');
            
            if (photos.length === 0) {
                container.innerHTML = '';
                emptyState.style.display = 'block';
                return;
            }
            
            emptyState.style.display = 'none';
            container.innerHTML = '';
            
            photos.forEach((photo, index) => {
                const photoCard = document.createElement('div');
                photoCard.className = 'photo-card';
                
                photoCard.innerHTML = `
                    <img src="${photo.data}" alt="乡村照片" class="photo-img">
                    <div class="photo-info">
                        <div class="photo-date">${formatDate(photo.date)}</div>
                        <div class="photo-actions">
                            <span>${photo.name}</span>
                            <button class="delete-btn" data-index="${index}">🗑️</button>
                        </div>
                    </div>
                `;
                
                container.appendChild(photoCard);
            });
            
            // 添加删除按钮事件
            document.querySelectorAll('.delete-btn').forEach(button => {
                button.addEventListener('click', function() {
                    const index = parseInt(this.getAttribute('data-index'));
                    deletePhoto(index);
                });
            });
        }
        
        // 处理文件上传
        function handleFileUpload(event) {
            const files = event.target.files;
            
            if (files.length === 0) return;
            
            for (let i = 0; i < files.length; i++) {
                const file = files[i];
                
                if (!file.type.match('image.*')) {
                    alert('请选择图片文件');
                    continue;
                }
                
                const reader = new FileReader();
                
                reader.onload = function(e) {
                    const photo = {
                        id: Date.now() + i,
                        name: file.name,
                        data: e.target.result,
                        date: new Date().toISOString(),
                        size: file.size
                    };
                    
                    photos.push(photo);
                    savePhotos();
                    displayPhotos();
                };
                
                reader.readAsDataURL(file);
            }
            
            // 重置文件输入
            event.target.value = '';
        }
        
        // 删除照片
        function deletePhoto(index) {
            if (confirm('确定要删除这张照片吗？')) {
                photos.splice(index, 1);
                savePhotos();
                displayPhotos();
            }
        }
        
        // 清空所有照片
        function clearAllPhotos() {
            if (photos.length === 0) {
                alert('相册已经是空的');
                return;
            }
            
            if (confirm('确定要清空整个相册吗？此操作不可撤销。')) {
                photos = [];
                savePhotos();
                displayPhotos();
            }
        }
        
        // 更新统计信息
        function updateStats() {
            document.getElementById('totalPhotos').textContent = photos.length;
            
            // 计算相册总大小
            let totalSize = 0;
            photos.forEach(photo => {
                totalSize += photo.size || 0;
            });
            
            document.getElementById('albumSize').textContent = formatFileSize(totalSize);
            
            // 更新最后更新时间
            if (photos.length > 0) {
                const lastPhoto = photos.reduce((latest, photo) => {
                    return new Date(photo.date) > new Date(latest.date) ? photo : latest;
                });
                
                document.getElementById('lastUpdate').textContent = formatDate(lastPhoto.date, true);
            } else {
                document.getElementById('lastUpdate').textContent = '-';
            }
        }
        
        // 格式化日期
        function formatDate(dateString, short = false) {
            const date = new Date(dateString);
            
            if (short) {
                return date.toLocaleDateString('zh-CN');
            }
            
            return date.toLocaleString('zh-CN');
        }
        
        // 格式化文件大小
        function formatFileSize(bytes) {
            if (bytes === 0) return '0 B';
            
            const k = 1024;
            const sizes = ['B', 'KB', 'MB', 'GB'];
            const i = Math.floor(Math.log(bytes) / Math.log(k));
            
            return parseFloat((bytes / Math.pow(k, i)).toFixed(2)) + ' ' + sizes[i];
        }
